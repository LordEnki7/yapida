function dispatchOrder(order, drivers) {
  const nearbyDrivers = drivers.filter(driver => {
    return getDistance(driver.location, order.pickup) < 5; // 5km radius
  });

  const scoredDrivers = nearbyDrivers.map(driver => {
    const distance = getDistance(driver.location, order.pickup);
    const eta = estimateETA(driver.location, order.pickup);

    const score =
      (0.4 * distance) +
      (0.3 * eta) -
      (0.2 * driver.rating) -
      (0.1 * driver.acceptanceRate);

    return { driver, score };
  });

  const rankedDrivers = scoredDrivers.sort((a, b) => a.score - b.score);

  const topDrivers = rankedDrivers.slice(0, 5);

  sendOrderRequest(topDrivers, order);
}
