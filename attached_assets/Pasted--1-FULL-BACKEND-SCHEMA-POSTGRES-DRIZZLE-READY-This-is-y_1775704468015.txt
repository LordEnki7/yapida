🧱 1. FULL BACKEND SCHEMA (POSTGRES + DRIZZLE READY)

This is your foundation. Don’t cut corners here.

👤 USERS TABLE
CREATE TABLE users (
  id UUID PRIMARY KEY,
  name TEXT,
  email TEXT UNIQUE,
  phone TEXT,
  role TEXT, -- customer | driver | business | admin
  created_at TIMESTAMP DEFAULT NOW()
);
🏪 BUSINESSES TABLE
CREATE TABLE businesses (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  name TEXT,
  category TEXT, -- food, supermarket, pharmacy, liquor
  address TEXT,
  lat FLOAT,
  lng FLOAT,
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMP DEFAULT NOW()
);
🛵 DRIVERS TABLE
CREATE TABLE drivers (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  is_online BOOLEAN DEFAULT false,
  rating FLOAT DEFAULT 5.0,
  acceptance_rate FLOAT DEFAULT 1.0,
  current_lat FLOAT,
  current_lng FLOAT,
  cash_balance FLOAT DEFAULT 0,
  wallet_balance FLOAT DEFAULT 0,
  created_at TIMESTAMP DEFAULT NOW()
);
📦 ORDERS TABLE
CREATE TABLE orders (
  id UUID PRIMARY KEY,
  customer_id UUID REFERENCES users(id),
  business_id UUID REFERENCES businesses(id),
  driver_id UUID REFERENCES drivers(id),

  status TEXT, -- pending, accepted, picked_up, delivered, cancelled

  total_amount FLOAT,
  delivery_fee FLOAT,
  commission FLOAT,

  payment_method TEXT, -- cash | card
  is_paid BOOLEAN DEFAULT false,

  pickup_lat FLOAT,
  pickup_lng FLOAT,
  dropoff_lat FLOAT,
  dropoff_lng FLOAT,

  created_at TIMESTAMP DEFAULT NOW()
);
🧾 ORDER ITEMS
CREATE TABLE order_items (
  id UUID PRIMARY KEY,
  order_id UUID REFERENCES orders(id),
  product_name TEXT,
  quantity INT,
  price FLOAT
);
💰 DRIVER WALLET TRANSACTIONS
CREATE TABLE driver_wallet (
  id UUID PRIMARY KEY,
  driver_id UUID REFERENCES drivers(id),
  type TEXT, -- earning | cash_collected | adjustment
  amount FLOAT,
  created_at TIMESTAMP DEFAULT NOW()
);
💬 CHAT / WHATSAPP LOG
CREATE TABLE messages (
  id UUID PRIMARY KEY,
  sender_id UUID,
  receiver_id UUID,
  message TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

🧑‍💻 DRIVER LOCATION UPDATE

Driver sends location every 3–5 seconds:

socket.emit("driverLocationUpdate", {
  driverId: "123",
  lat: 18.4861,
  lng: -69.9312
});
🧠 SERVER HANDLER
io.on("connection", (socket) => {

  socket.on("driverLocationUpdate", async (data) => {
    const { driverId, lat, lng } = data;

    await db.update(drivers)
      .set({ current_lat: lat, current_lng: lng })
      .where(eq(drivers.id, driverId));

    io.emit(`trackDriver_${driverId}`, { lat, lng });
  });

});
📱 CUSTOMER TRACKING
socket.on(`trackDriver_${driverId}`, (location) => {
  updateMap(location.lat, location.lng);
});

app.post("/admin/lock-driver", async (req, res) => {
  const { driverId } = req.body;

  await db.update(drivers)
    .set({ is_online: false })
    .where(eq(drivers.id, driverId));

  res.send("Driver locked");
});
